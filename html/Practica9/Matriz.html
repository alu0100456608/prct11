def == (other)
   @filas.times do |i|
      @columnas.times do |j|
         if at(i, j) != other.at(i,j)
            return false
         end
      end
   end
   
   return true
end

# File lib/practica9.rb, line 16
def to_s
   tmp = ""
   @filas.times do |i|
      @columnas.times do |j|
         tmp << "#{at(i, j)} "
      end
      tmp << "\n"
   end
   tmp
end
